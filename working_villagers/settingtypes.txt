[debugging]

#    Enables the debug tools for working villages.
#    They can be obtained ingame through the creative inventory.
#    They affect the nearby nodes in some way that helps to see if working villages is working properly.
working_villages_enable_debug_tools (debug tools) bool false

#    Enables some extra asserts in the code that ensure everything runs as expected.
#    This may have a small effect on the performance.
#    It helps finding bugs within working villages and within mods using working villages.
working_villages_enable_debug_checks (debug checks) bool true

[protection]
#    Controls activity in protected areas.
#    false: Activity is not allowed in protected areas.
#    true: Activity is allowed in protected areas if the owner would be allowed.
#    ignore: Activity is allowed in protected areas.
#    prefix: Protection is tested using a fake player name of "[prefix]:[owner]"
#            For example, if the owner is "user1" and the prefix is "owner"
#            the test will use the player name "owner:user1". This is
#            useful with protection mods that allow protection to be
#            configured using arbitrary player names.
working_villages_owner_protection (owner protection) string false

[spawning]

#    Enables spawning self-employed working villagers. The villagers that
#    spawn in will interact with any player who has a command sceptre.
working_villages_enable_spawn (spawn working villagers) bool false

[autonomous]

#    Minimum interval in seconds between villager chat messages.
working_villages_autonomous_chat_interval (autonomous chat interval) int 30

#    Radius (nodes) used for autonomous exploration.
working_villages_autonomous_explore_radius (autonomous explore radius) int 10

[builder]

#    Builders get missing materials automatically.
working_villages_builder_unlimited_materials (builder unlimited materials) bool true

#    Search radius (nodes) for construction markers.
working_villages_builder_search_radius (builder search radius) int 30

#    Builders attempt experimental builds when possible.
working_villages_builder_experiment_mode (builder experiment mode) bool true

#    Time (seconds) between experimental build attempts.
working_villages_builder_experiment_interval (builder experiment interval) int 1800

[guard]

#    Default guard mode: stationary, escort, patrol, wandering.
working_villages_guard_default_mode (guard default mode) string patrol

#    Patrol radius (nodes) for guard mode "patrol".
working_villages_guard_patrol_radius (guard patrol radius) int 12

#    Guards equip a weapon automatically if they have none.
working_villages_guard_auto_weapon (guard auto weapon) bool true
